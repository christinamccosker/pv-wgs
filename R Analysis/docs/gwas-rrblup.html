6.3 Calculating GRM

6.3.1 Extract marker calls from snp matrix

SNP calls need to be changed from 0, 1, 2 to -1, 0, 1 (for rrBLUP)

x <- 
  snp_matrix[, -c(1:6)] %>% 
  as.matrix()

x <- x-1

colnames(x) <- gsub("\\_[A-Z]{1}$","",colnames(x))

rownames(x) <- snp_matrix$IID

6.3.2 Calculate GRM

grm <- A.mat(x)

6.4 Check for population structure

6.4.1 Heatmap

No obvious clustering of samples

heatmap(grm,labRow = FALSE, labCol = FALSE, col=rev(heat.colors(75)))

6.4.2 PCA

Again, seemingly no sample clustering

pca <- prcomp(grm, scale=TRUE, center=TRUE)

autoplot(pca, data=meta, colour="classification") +
  theme_bw() +
  theme(panel.grid=element_blank())

6.4.2.1 Variance explained by first 8 PCs

Relatively low variance explained

eigenvalues <- pca$sdev^2
round((eigenvalues/sum(eigenvalues)*100), 2)[1:8] 

for(i in 1:8){
  varExp <- (eigenvalues/sum(eigenvalues)*100)[1:i]
  print(paste0("Var explained by first ", i, " PCs: ", round(sum(varExp), 2),
               "%"))
}

fviz_eig(pca, ncp = 8)

6.5 Prepare data for rrBLUP

6.5.1 Create matrix

Add marker location and marker dosage

snp_info <- 
  bind_cols(snp_info, as.data.frame(t(x)))

6.5.2 Configure phenotype data

Non-survivor = 0, Survivor = 1

phenotypes <- 
  meta %>% 
  select(seq.id, classification) %>% 
  setNames(c("id", "phenotype")) %>% 
  mutate(id = gsub("B5GAN", "", id),
         phenotype = ifelse(phenotype=="Non-survivor", 0, 1))

# Make sure sample ID order matches
phenotypes <- 
  phenotypes[order(match(phenotypes$id, rownames(x))), ]

phenotypes$id == rownames(x)

6.5.3 Configure grm & snp_info

vec <- colnames(grm) %in% phenotypes$id
grm <- grm[vec,vec]

snp_info <- as.data.frame(snp_info)
snp_info <- snp_info[,c(TRUE,TRUE,TRUE,vec)]

6.7.1 Run model

gwasResults_1 <-
  model_1 %>%
