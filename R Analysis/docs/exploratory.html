2.4 Map of Strandings

usa <- st_as_sf(maps::map("state", fill=TRUE, plot =FALSE))

map <- ggplot(usa) +
    geom_sf() + 
    geom_point(data=meta, aes(x=long, y=lat, color=pdv, shape=pdv), size=2.5) +
    scale_color_manual(values=c("black","slategray3", "#58baba"), name="PDV Status", 
                       labels=c("Unknown", "Negative", "Positive")) +
    scale_shape_manual(values=c(15, 18, 16),  name="PDV Status", 
                       labels=c("Unknown", "Negative", "Positive")) +
    coord_sf(xlim=c(-67.3,-71.8), ylim=c(41,44.8)) +
    theme_bw() +
    theme(text = element_text(size=20)) +
    labs(x=NULL, y=NULL)

2.5 Case v Control ~ Factors

2.5.1 During/After Outbreak

outbreak <-
  meta %>% 
  select(classification, pdv) %>% 
  mutate(pdv=factor(pdv, levels=c("Positive", "Negative", "N/A"))) %>% 
  ggplot(data=.) +
  geom_mosaic(aes(x=product(pdv,classification), fill=pdv)) +
  scale_fill_manual(values=c("N/A" = "grey",
                             "Negative" = "#008080",
                             "Positive" = "#58baba"),
                    name="PDV") +
  scale_y_continuous(labels = scales::percent) +
  labs(y="Proportion of Samples") +
  theme_bw() +
  theme(panel.grid.major = element_blank(), 
        panel.grid.minor = element_blank(),
        axis.title.x = element_blank(),
        legend.position="bottom",
        text = element_text(size=10),
        plot.title = element_text(hjust = 0.5))
outbreak
